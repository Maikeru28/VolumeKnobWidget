# Runtime dependencies for the Volume Knob application
# Pin versions here if you want fully reproducible installs. For now we keep them unpinned
# so the latest compatible versions are fetched.

PyQt6          # GUI framework
pycaw          # Windows Core Audio (endpoint + session control)
comtypes       # COM interop (explicit, even though pycaw depends on it)
psutil         # Process info objects returned in audio sessions (dependency of pycaw)
